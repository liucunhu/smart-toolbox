## 🔍 代码示例

### 合规审查对话框组件

```vue
<!-- src/components/ComplianceCheckDialog.vue -->
<template>
  <el-dialog
    v-model="visible"
    title="合规审查结果"
    width="600px"
  >
    <div v-if="!passed" class="compliance-failed">
      <el-alert
        type="error"
        :title="errorMessage"
        show-icon
      />
      
      <div class="violations-list">
        <h4>检测到以下违禁词：</h4>
        <el-tag
          v-for="(word, index) in violations"
          :key="index"
          type="danger"
          style="margin: 5px"
        >
          {{ word }}
        </el-tag>
      </div>
      
      <div class="suggestions">
        <h4>修改建议：</h4>
        <ul>
          <li>请使用更客观的描述</li>
          <li>避免使用绝对化用语</li>
          <li>参考平台内容规范</li>
        </ul>
      </div>
    </div>
    
    <div v-else class="compliance-passed">
      <el-alert
        type="success"
        title="✅ 合规审查通过"
        show-icon
      />
    </div>
    
    <template #footer>
      <el-button @click="visible = false">关闭</el-button>
      <el-button
        v-if="!passed"
        type="primary"
        @click="$emit('retry')"
      >
        重新检查
      </el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref } from 'vue'

const visible = ref(false)
const passed = ref(true)
const errorMessage = ref('')
const violations = ref([])

const showResult = (result) => {
  passed.value = result.passed
  errorMessage.value = result.error || ''
  violations.value = result.violations || []
  visible.value = true
}

defineExpose({ showResult })
defineEmits(['retry'])
</script>
```
